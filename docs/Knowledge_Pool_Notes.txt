def _extract_added_code_by_file(diff_text):
    file_code_map = {}
    current_file = None
    in_hunk = False
    for line in diff_text.splitlines():
        if line.startswith("diff --git "):
            current_file = None
            in_hunk = False
            continue
        if line.startswith("+++ "):
            if line.startswith("+++ b/"):
                current_file = line[6:]
            elif line == "+++ /dev/null":
                current_file = None
            continue
        if line.startswith("@@"):
            in_hunk = True
            continue
        if not in_hunk:
            continue
        if line.startswith("+"):
            added_line = line[1:]
            if current_file and added_line.strip():
                file_code_map.setdefault(current_file, []).append(added_line)
    return file_code_map
